32       @Test(dataProvider = "csv")
33       public void test(Map<String, String> map) {
34           String url = "/api/emissions-mgmt/apportioned/hourly/?page=%s&perPage=%s&beginDate=%s&endDate=%s&orisCode=%s|%s";
35           String oris1 = "127";
36           String oris2 = "612";
37   //        Perform a GET request for ORIS = 127 and 612
38   //        https://easey-dev.app.cloud.gov/api/emissions-mgmt/apportioned/hourly/?page=1&perPage=3000&beginDate=2019-01-01&endDate=2019-01-31&orisCode=127|612
39           JSONArray response = getJSONArray(String.format(url, "1", map.get("perPage"), map.get("beginDate"), map.get("endDate"), oris1, oris2));
40   //        API returns results for both ORIS 127 and 612
41           for (Object r : response) {
42               if (r instanceof JSONObject) {
43                   JSONObject res = (JSONObject) r;
44                   String oris = res.getString("orisCode");
45                   verifyTrue(oris.equals(oris1) || oris.equals(oris2),
46                           "Response was not from the correct ORIS Code\n" + res.toString(4));
47               } else
48                   verifyFail("Response returned non JSONObject\n" + r.toString());
49           }
50   
51   //        Perform a GET request to test allowing white space
52   //        https://easey-dev.app.cloud.gov/api/emissions-mgmt/apportioned/hourly/?page=1&perPage=3000&beginDate=2019-01-01&endDate=2019-01-31&orisCode=127%20|%20612
53           Response rRes = getResponse(String.format(url, "1", map.get("perPage"), map.get("beginDate"), map.get("endDate"), oris1 + " ", oris2 + " "));
54   //        API returns results for both ORIS 127 and 612
55           if (rRes.getStatusCode() == 200) {
56               response = new JSONArray(rRes.getBody().asString());
57               for (Object r : response) {
58                   if (r instanceof JSONObject) {
59                       JSONObject res = (JSONObject) r;
60                       String oris = res.getString("orisCode");
61                       verifyTrue(oris.equals(oris1) || oris.equals(oris2),
62                               "Response was not from the correct ORIS Code\n" + res.toString(4));
63                   } else
64                       verifyFail("Response returned non JSONObject\n" + r.toString());
65               }
66           } else {
67               verifyFail("Invalid Response Code: " + rRes.getStatusCode());
68           }
69       }
